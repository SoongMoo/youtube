create table survey(
	survey_num number,
	survey_writer varchar2(200),
	survey_subject varchar2(200),
	survey_start timestamp,
    survey_end timestamp
);

create table question(
    survey_num number,
    survey_writer varchar2(200),
    question_num number,
    question_title VARCHAR2(500)
);

create table options(
    survey_num number,
    survey_writer varchar2(200),
    Question_num number,
    options_num number,
    options_content varchar2(2000)
);
